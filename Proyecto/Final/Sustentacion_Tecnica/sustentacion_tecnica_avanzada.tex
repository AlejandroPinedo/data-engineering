% --- Slide 5: Caso Técnico Cuarentena ---
\begin{frame}[fragile]{Caso Técnico: Cuarentena y la Falla de Ceros a la Izquierda}
    \begin{columns}
        \begin{column}{0.48\textwidth}
            \textbf{Diagnóstico del Incidente de 233.69K registros:}
            \begin{itemize}
                \setlength{\itemsep}{0.4em}
                \item \textbf{El Síntoma:} La tabla de cuarentena acumulaba toda la data y Gold quedaba en blanco.
                \item \textbf{La Causa:} El código de material de SAP tenía ceros a la izquierda (`000000000000100243`), mientras que el maestro de materiales no (`100243`). El left join fallaba, resultando en Proveedor nulo y gatillando el filtro de cuarentena.
                \item \textbf{La Solución:} Remover ceros de forma simétrica con \texttt{ltrim(col, '0')}.
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{lstlisting}[language=Python, caption=Resolucion de Clave de Join e Ingesta de Cuarentena]
from pyspark.sql.functions import ltrim, col

# Estandarizar clave de join
df_mov_clean = df_mov_clean.withColumn("Material_Join", ltrim(col("Material"), "0"))
df_maestro_clean = df_maestro.withColumn("Material_Join", ltrim(col("Material"), "0"))

df_joined = df_mov_clean.join(df_maestro_clean, "Material_Join", "left")

# Regla de desvio a Cuarentena
condicion_cuarentena = (col("Qty").isNull()) | (col("Qty") == 0) | \
                       (col("Proveedor").isNull()) | (col("Posting_Date").isNull())

df_joined.filter(condicion_cuarentena).select("Material_Id", "Posting_Date", "Qty") \
  .write.mode("append").saveAsTable("silver.silver_movimientos_cuarentena")
            \end{lstlisting}
        \end{column}
    \end{columns}
\end{frame}

    \begin{columns}

            \textbf{Vistas Semánticas de Negocio:}

                \item \textbf{DP1:} Alertas de reorden automáticas al caer el stock acumulado por debajo del mínimo.
                \item \textbf{DP2:} Consumos netos en USD agrupados por mes y categoría.
            \end{itemize}

        \begin{column}{0.5\textwidth}
            \begin{lstlisting}[language=SQL, caption=Lógica SQL de Alertas e Inventarios (DP1)]
CREATE OR REPLACE VIEW gold.mv_replenishment_alerts AS
WITH stock_acumulado AS (
  SELECT material_id, descripcion, proveedor, categoria, stock_seguridad, 
         punto_reorden, costo_unitario_usd, SUM(cantidad) AS stock_actual
  FROM silver.silver_movimientos_consignacion
  GROUP BY 1, 2, 3, 4, 5, 6, 7
)
SELECT *,
  CASE 
    WHEN stock_actual < stock_seguridad THEN 'CRITICO - DESABASTECIMIENTO'
    WHEN stock_actual < punto_reorden THEN 'REORDEN SUGERIDO'
    ELSE 'NORMAL'
  END AS estado_inventario,
  (punto_reorden - stock_actual) AS cantidad_a_comprar,
  ((punto_reorden - stock_actual) * costo_unitario_usd) AS costo_reposicion_usd
FROM stock_acumulado
WHERE stock_actual < punto_reorden;
            \end{lstlisting}
        \end{column}
    \end{columns}
\end{frame}

CREATE OR REPLACE VIEW g102_fin_conciliacion.gold.mv_monthly_conciliation AS
SELECT 
  date_format(fecha_movimiento, 'yyyy-MM') AS anio_mes, 
  proveedor, 
  documento_material, 
  material_id, 
  descripcion, 
  costo_unitario_usd,
  ABS(SUM(cantidad)) AS cantidad_facturable,
  ABS(SUM(monto_total_usd)) AS monto_facturable_usd
FROM g102_log_reabastecimiento.silver.silver_movimientos_consignacion
WHERE cantidad < 0
GROUP BY 1, 2, 3, 4, 5, 6;
