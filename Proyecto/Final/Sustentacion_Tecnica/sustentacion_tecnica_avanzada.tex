# Mapear signo según tipo de movimiento SAP

% --- Slide 5: El Caso de la Cuarentena ---
\begin{frame}[fragile]{Caso Técnico: Cuarentena de 233.69K Registros}
    \begin{columns}
        \begin{column}{0.48\textwidth}
            \textbf{Diagnóstico de la Falla del Join:}
            \begin{itemize}
                \setlength{\itemsep}{0.3em}
                \item \textbf{El Síntoma:} 233.69K registros caían en la tabla de cuarentena y los Data Products quedaban vacíos.
                \item \textbf{La Causa:} SAP almacena códigos de material con ceros a la izquierda (ej. `000000000000100243`), mientras que el maestro de materiales los tenía sin ceros (`100243`). El join fallaba, y el campo `Proveedor` era nulo.
                \item \textbf{La Solución:} Aplicar \texttt{ltrim(col, "0")} en ambas fuentes antes de unir.
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{lstlisting}[language=Python, caption=Solución al Join con ltrim]
from pyspark.sql.functions import ltrim, col

# Estandarizar claves eliminando ceros a la izquierda
df_mov_clean = df_mov_clean.withColumn(
    "Material_Join", 
    ltrim(col("Material"), "0")
)
df_maestro_clean = df_maestro.withColumn(
    "Material_Join", 
    ltrim(col("Material"), "0")
)

# Join por clave unificada para evitar proveedor nulo
df_joined = df_mov_clean.join(
    df_maestro_clean, 
    "Material_Join", 
    "left"
)
            \end{lstlisting}
        \end{column}
    \end{columns}
\end{frame}

SELECT 
  date_format(fecha_movimiento, 'yyyy-MM') AS anio_mes, 
  proveedor, 
  documento_material, 
  material_id, 
  descripcion, 
  costo_unitario_usd,
  ABS(SUM(cantidad)) AS cantidad_facturable,
  ABS(SUM(monto_total_usd)) AS monto_facturable_usd
